# --- HELPER: DETECT OS & ENVIRONMENT ---
get_os_env() {
    # Detect OS (OEL8, RHEL9, etc)
    OS_TYPE=$(hostnamectl | grep "Operating System" | awk -F: '{print $2}' | awk '{type=($1=="Oracle"?"OEL":"RHEL"); match($0,/[0-9]/); print type substr($0,RSTART,1)}')
    
    # Identify Env (10.x = PROD, 172.x = TEST)
    MAIN_IP=$(hostname -I | awk '{print $1}')
    if [[ $MAIN_IP =~ ^10\. ]]; then
        ENV="PROD"
    elif [[ $MAIN_IP =~ ^172\. ]]; then
        ENV="TEST"
    else
        ENV="UNKNOWN"
    fi
    
    KEY="${OS_TYPE}_${ENV}"
    REPO_INFO="${HOSTS_DATA[$KEY]}"
    
    if [ -z "$REPO_INFO" ]; then
        echo "Error: No mapping found for $KEY"
        exit 1
    fi
    
    REPO_IP=$(echo $REPO_INFO | awk '{print $1}')
    REPO_HOST=$(echo $REPO_INFO | awk '{print $2}')
    REPO_FQDN=$(echo $REPO_INFO | awk '{print $3}')
}

# --- FUNCTION: CHECK CONNECTION ---
check_connection() {
    get_os_env
    PORT=443
    echo "Checking connection to $REPO_HOST ($REPO_IP) on port $PORT..."
    
    if command -v nc >/dev/null; then
        nc -zv -w3 "$REPO_IP" "$PORT" >/dev/null 2>&1
    else
        (echo "quit"; sleep 1) | telnet "$REPO_IP" "$PORT" 2>&1 | grep -q "Connected"
    fi

    if [ $? -eq 0 ]; then
        echo "Connected | Local Repo | $REPO_IP:$PORT"
        return 0
    else
        echo "Could not Connect to Local Repo ($REPO_IP). Please ask network team for access."
        return 1
    fi
}

# --- FUNCTION: CHECK/ADD HOST ENTRY ---
check_hostentry() {
    get_os_env
    # Check if IP exists (even if commented)
    if grep -q "$REPO_IP" /etc/hosts; then
        if grep -q "^#.*$REPO_IP" /etc/hosts; then
            echo "Uncommenting existing entry for $REPO_IP..."
            sed -i "/$REPO_IP/s/^#//" /etc/hosts
        else
            echo "Host entry for $REPO_IP is already active."
        fi
    else
        echo "Adding host entry: $REPO_IP $REPO_HOST $REPO_FQDN"
        echo -e "$REPO_IP\t$REPO_HOST\t$REPO_FQDN" >> /etc/hosts
    fi
}
